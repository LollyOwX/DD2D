        public Font setup(String name, String style) {
            int fontStyle;
            switch(style.toUpperCase()) {
                case "P": fontStyle = Font.PLAIN; break;
                case "B": fontStyle = Font.BOLD; break;
                case "I": fontStyle = Font.ITALIC; break;
                case "BI":
                case "IB": fontStyle = Font.BOLD | Font.ITALIC; break;
                default: fontStyle = Font.PLAIN;
            }
            return new Font(name, fontStyle, 1); // dimensione provvisoria
        }
